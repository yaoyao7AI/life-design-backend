- `enableKeepAlive: true`（启用 keep-alive）

## 🚀 部署步骤

### 在服务器上执行：

```bash
cd /root/apps/life-design-backend

# 备份原文件
cp src/db.js src/db.js.backup.$(date +%Y%m%d_%H%M%S)

# 更新文件
cat > src/db.js << 'EOF'
import mysql from "mysql2/promise";
import dotenv from "dotenv";

// 确保环境变量已加载（server.js 已经加载，这里再次确保）
dotenv.config();

// 添加调试日志，确认环境变量是否正确加载
console.log('[DB] 创建连接池，DB_HOST:', process.env.DB_HOST);
console.log('[DB] DB_PORT:', process.env.DB_PORT);
console.log('[DB] DB_USER:', process.env.DB_USER);
console.log('[DB] DB_NAME:', process.env.DB_NAME);

export const pool = mysql.createPool({
  host: process.env.DB_HOST,
  port: Number(process.env.DB_PORT),
  user: process.env.DB_USER,
  password: process.env.DB_PASS,
  database: process.env.DB_NAME,
  connectionLimit: 10,
  connectTimeout: 30000, // 增加连接超时时间到 30 秒
  acquireTimeout: 30000, // 获取连接的超时时间
  timeout: 30000, // 查询超时时间
  enableKeepAlive: true, // 启用 keep-alive
  keepAliveInitialDelay: 0, // keep-alive 初始延迟
});
EOF

# 验证文件内容
echo "=== 验证更新后的配置 ==="
cat src/db.js | grep -A 10 "createPool"

# 重启服务
echo ""
echo "=== 重启服务 ==="
pm2 stop life-design-backend
pm2 delete life-design-backend
pm2 save --force
sleep 3
pm2 start src/server.js --name life-design-backend --update-env
sleep 5
pm2 logs life-design-backend --lines 30 --nostream
```
